FUNCTION wordCounter(paragraph):
    // Step 1: Normalize text
    Convert paragraph to lowercase
    Remove punctuation (.,!? etc.)

    // Step 2: Split words
    words = split paragraph by spaces

    // Step 3: Count frequency
    CREATE Map<String, Integer> wordCount

    FOR each word in words:
        IF wordCount contains word
            wordCount[word] = wordCount[word] + 1
        ELSE
            wordCount[word] = 1

    // Step 4: Sort by frequency
    sortedList = wordCount entries sorted in descending order of value

    // Step 5: Display result
    FOR each entry in sortedList:
        PRINT entry.key + " â†’ " + entry.value
END FUNCTION


MAIN PROGRAM:
    paragraph = "The news today is about the rise in technology trends."

    CALL wordCounter(paragraph)
