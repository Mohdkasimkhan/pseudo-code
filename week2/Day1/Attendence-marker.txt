CLASS AttendanceSystem
    ATTRIBUTES:
        attendance[students][days]   // 2D Array (rows = students, cols = days)

    METHOD markAttendance(studentId, day, status):
        attendance[studentId][day] = status
        PRINT "Attendance marked for Student " + studentId + " on Day " + day

    METHOD updateAttendance(studentId, day, newStatus):
        attendance[studentId][day] = newStatus
        PRINT "Attendance updated for Student " + studentId + " on Day " + day

    METHOD generateReport(studentId, totalDays):
        presentCount = 0
        FOR day = 1 TO totalDays
            IF attendance[studentId][day] == 'P'
                presentCount++
        PRINT "Student " + studentId + " was present for " + presentCount + " days."
END CLASS


MAIN PROGRAM:
    totalStudents = 3
    totalDays = 30
    CREATE AttendanceSystem with [totalStudents][totalDays]

    // Mark attendance
    CALL markAttendance(1, 1, 'P')
    CALL markAttendance(1, 2, 'P')
    CALL markAttendance(1, 3, 'A')
    CALL markAttendance(1, 4, 'P')
    CALL markAttendance(1, 5, 'P')
    CALL markAttendance(1, 6, 'P')
    CALL markAttendance(1, 7, 'A')

    // Teacher updates day 3 from A â†’ P
    CALL updateAttendance(1, 3, 'P')

    // Generate report for Student 1
    CALL generateReport(1, totalDays)
