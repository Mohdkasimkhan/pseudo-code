class Flight {
    String flightNo;
    int price;
    int duration;
}

class FlightComparator implements Comparator<Flight> {
    int compare(Flight f1, Flight f2) {
        if (f1.price != f2.price) return f1.price - f2.price;
        return f1.duration - f2.duration;
    }
}
